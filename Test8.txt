10 LET N = 120
20 LET M = 240
30 LET S1 = 1
40 LET S2 = 1
50 GOSUB 1050
60 GOSUB 2050
70 IF S1 <> M GOTO 110
80 IF S2 <> N GOTO 110
90 LET F = 1
100 GOTO 120
110 LET F = 0
120 GOTO 5000
1000 REM calculating sum of dividers for N
1050 FOR C1 = 2 TO N - 1 STEP 1
1060 GOSUB 3000
1070 IF T1 = 1 GOTO 1090
1080 NEXT C1
1090 LET S1 = S1 + C1
1100 GOTO 1080
1110 RETURN
2000 REM calculating sum of dividers for M
2050 FOR C2 = 2 TO M - 1 STEP 1
2060 GOSUB 4000
2070 IF T2 = 1 GOTO 2090
2080 NEXT C2
2090 LET S2 = S2 + C2
2100 GOTO 2080
2110 RETURN
3000 LET D = 0
3010 FOR D = 0 TO N STEP 1
3020 LET P = D * C1
3030 IF P >= N GOTO 3050
3040 NEXT D
3050 IF P = N GOTO 3080
3060 LET T1 = 0
3070 RETURN
3080 LET T1 = 1
3090 RETURN
4000 LET D = 0
4010 FOR D = 0 TO N STEP 1
4020 LET P = D * C2
4030 IF P >= M GOTO 4050
4040 NEXT D
4050 IF P = M GOTO 4080
4060 LET T2 = 0
4070 RETURN
4080 LET T2 = 1
4090 RETURN
5000 END
